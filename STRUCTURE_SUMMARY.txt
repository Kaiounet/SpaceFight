    while (!WindowShouldClose) {
        // 1. NETWORK - Process received messages
        while (hasMessages) {
            msg = pollMessage()
            case PLAYER_JOIN:
                players.put(msg.playerId, new Player(...))
            case PLAYER_MOVE:
                players.get(msg.playerId).move(msg.x, msg.y)
            case PLAYER_LEAVE:
                players.remove(msg.playerId)
        }
        
        // 2. INPUT - Handle local player movement
        dx = dy = 0
        if (KEY_LEFT || KEY_A):  dx = -5
        if (KEY_RIGHT || KEY_D): dx = +5
        if (KEY_UP || KEY_W):    dy = -5
        if (KEY_DOWN || KEY_S):  dy = +5
        
        if (dx != 0 || dy != 0):
            newX = clamp(localPlayer.x + dx, 0, WIDTH)
            newY = clamp(localPlayer.y + dy, 0, HEIGHT)
            
            // 3. SEND - Broadcast to all players
            msg = PLAYER_MOVE(playerId, newX, newY)
            client.sendMessage(msg)  // async background thread
            
            localPlayer.move(newX, newY)
        
        // 4. RENDER - Draw everything
        BeginDrawing()
        ClearBackground(DARKGRAY)
        
        for each player in players:
            DrawRectangle(player.x, player.y, SIZE, SIZE, player.color)
            DrawText("P" + player.id, ...)
        
        DrawText("Players: " + players.size(), ...)
        EndDrawing()
    }
